function fetchPicture() { const apiKey = document.getElementById("apiKey").value.trim(); const date = document.getElementById("datePicker").value; const output = document.getElementById("output"); if (!apiKey || !date) { alert("Please enter both API key and date."); return; } const apiUrl = `https://api.nasa.gov/planetary/apod?api_key=${apiKey}&date=${date}`; fetch(apiUrl) .then(response => response.json()) .then(data => { if (data.media_type === "image") { output.innerHTML = `

${data.title}

Date: ${data.date}

${data.explanation}

`; } else { output.innerHTML = `

This date's content is not an image. Try another one.

`; } }) .catch(error => { console.error(error); output.innerHTML = `

Error fetching data. Check your API key and try again.

`; }); }
