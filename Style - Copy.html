document.getElementById("loadBtn").addEventListener("click", function () { const apiKey = document.getElementById("apiKey").value.trim(); const date = document.getElementById("datePicker").value; if (!apiKey || !date) { alert("Please enter both API key and date."); return; } sessionStorage.setItem("nasaApiKey", apiKey); const url = `https://api.nasa.gov/planetary/apod?api_key=${apiKey}&date=${date}`; fetch(url) .then(response => response.json()) .then(data => { const result = document.getElementById("result"); if (data.media_type === "image") { result.innerHTML = `

${data.title}

Date: ${data.date}

${data.explanation}

`; } else { result.innerHTML = `

This date's content is not an image. Please choose another date.

`; } }) .catch(error => { console.error("Error:", error); alert("Failed to fetch data. Please check your API key."); }); });
