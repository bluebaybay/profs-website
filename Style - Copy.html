document.getElementById("loadBtn").addEventListener("click", function () { const apiKey = document.getElementById("apiKey").value.trim(); const date = document.getElementById("datePicker").value; const result = document.getElementById("result"); if (!apiKey || !date) { alert("Please enter both your NASA API key and a valid date."); return; } sessionStorage.setItem("nasaApiKey", apiKey); const url = `https://api.nasa.gov/planetary/apod?api_key=${apiKey}&date=${date}`; fetch(url) .then(response => { if (!response.ok) { throw new Error("Network response was not OK"); } return response.json(); }) .then(data => { if (data.media_type === "image") { result.innerHTML = `

${data.title}

Date: ${data.date}

${data.explanation}

`; } else { result.innerHTML = `

This date's content is not an image. Please choose another date.

`; } }) .catch(error => { console.error("Error fetching NASA data:", error); result.innerHTML = `

Something went wrong. Please check your API key or try again later.

`; }); });
